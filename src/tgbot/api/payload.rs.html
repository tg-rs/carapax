1use std::{error::Error, fmt};
2
3use log::debug;
4use reqwest::{Client as HttpClient, Method as HttpMethod, RequestBuilder as HttpRequestBuilder};
5use serde::ser::Serialize;
6use serde_json::Error as JsonError;
7
8use super::form::{Form, FormError};
9
10#[derive(Debug)]
11pub(crate) enum PayloadData {
12    Form(Form),
13    Json(Result<String, JsonError>),
14    Empty,
15}
16
17#[doc(hidden)]
18#[derive(Debug)]
19pub struct Payload {
20    pub(crate) http_method: HttpMethod,
21    pub(crate) payload_data: PayloadData,
22    pub(crate) url_path: String,
23}
24
25impl Payload {
26    pub(crate) fn form<P: Into<String>>(path: P, form: Form) -> Self {
27        Self {
28            http_method: HttpMethod::POST,
29            payload_data: PayloadData::Form(form),
30            url_path: path.into(),
31        }
32    }
33
34    pub(crate) fn json<P: Into<String>>(path: P, data: impl Serialize) -> Self {
35        Self {
36            http_method: HttpMethod::POST,
37            payload_data: PayloadData::Json(serde_json::to_string(&data)),
38            url_path: path.into(),
39        }
40    }
41
42    pub(crate) fn empty<P: Into<String>>(path: P) -> Self {
43        Self {
44            http_method: HttpMethod::GET,
45            payload_data: PayloadData::Empty,
46            url_path: path.into(),
47        }
48    }
49
50    pub(crate) fn build_url(&self, base_url: &str, token: &str) -> String {
51        format!("{}/bot{}/{}", base_url, token, self.url_path)
52    }
53
54    pub(super) fn into_http_request_builder(
55        self,
56        http_client: &HttpClient,
57        base_url: &str,
58        token: &str,
59    ) -> Result<HttpRequestBuilder, PayloadError> {
60        let url = self.build_url(base_url, token);
61        let builder = http_client.request(self.http_method, url);
62        Ok(match self.payload_data {
63            PayloadData::Form(form) => {
64                let form = form.try_into()?;
65                debug!("Sending multipart body: {form:?}");
66                builder.multipart(form)
67            }
68            PayloadData::Json(data) => {
69                let data = data?;
70                debug!("Sending JSON body: {data:?}");
71                builder.header("Content-Type", "application/json").body(data)
72            }
73            PayloadData::Empty => {
74                debug!("Sending empty body");
75                builder
76            }
77        })
78    }
79}
80
81/// An error when building an HTTP request
82#[derive(Debug, derive_more::From)]
83pub enum PayloadError {
84    /// Failed to build a form body
85    Form(FormError),
86    /// Failed to build a JSON body
87    Json(JsonError),
88}
89
90impl Error for PayloadError {
91    fn source(&self) -> Option<&(dyn Error + 'static)> {
92        use self::PayloadError::*;
93        Some(match self {
94            Form(err) => err,
95            Json(err) => err,
96        })
97    }
98}
99
100impl fmt::Display for PayloadError {
101    fn fmt(&self, out: &mut fmt::Formatter) -> fmt::Result {
102        use self::PayloadError::*;
103        write!(
104            out,
105            "could not build an HTTP request: {}",
106            match self {
107                Form(err) => err.to_string(),
108                Json(err) => err.to_string(),
109            }
110        )
111    }
112}
