8/// Represents a position on faces where a mask should be placed by default.
9#[derive(Clone, Copy, Debug, Deserialize, PartialEq, PartialOrd, Serialize)]
10pub struct MaskPosition {
11    /// The part of the face relative
12    /// to which the mask should be placed.
13    pub point: MaskPositionPoint,
14    /// Mask scaling coefficient.
15    ///
16    /// For example, 2.0 means double size.
17    pub scale: Float,
18    /// Shift by X-axis measured in widths
19    /// of the mask scaled to the face size,
20    /// from left to right.
21    ///
22    /// For example, choosing -1.0
23    /// will place mask just
24    /// to the left of the default mask position.
25    pub x_shift: Float,
26    /// Shift by Y-axis measured
27    /// in heights of the mask scaled to the face size,
28    /// from top to bottom.
29    ///
30    /// For example, 1.0 will place
31    /// the mask just below the default mask position.
32    pub y_shift: Float,
33}
34
35impl MaskPosition {
36    /// Creates a new `MaskPosition`.
37    ///
38    /// # Arguments
39    ///
40    /// * `point` - The part of the face.
41    /// * `scale` - Mask scaling coefficient.
42    /// * `x_shift` - Shift by X-axis.
43    /// * `y_shift` - Shift by Y-axis.
44    pub fn new(point: MaskPositionPoint, scale: Float, x_shift: Float, y_shift: Float) -> Self {
45        Self {
46            point,
47            scale,
48            x_shift,
49            y_shift,
50        }
51    }
52}
