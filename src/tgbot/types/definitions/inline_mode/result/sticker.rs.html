11/// Represents a link to a sticker stored on the Telegram servers.
12///
13/// By default, this sticker will be sent by the user.
14/// Alternatively, you can use [`Self::with_input_message_content`] to
15/// send a message with the specified content instead of the sticker.
16#[serde_with::skip_serializing_none]
17#[derive(Clone, Debug, Deserialize, PartialEq, Serialize)]
18pub struct InlineQueryResultCachedSticker {
19    id: String,
20    sticker_file_id: String,
21    input_message_content: Option<InputMessageContent>,
22    reply_markup: Option<InlineKeyboardMarkup>,
23}
24
25impl InlineQueryResultCachedSticker {
26    /// Creates a new `InlineQueryResultCachedSticker`.
27    ///
28    /// # Arguments
29    ///
30    /// * `id` - Unique identifier of the result; 1-64 bytes.
31    /// * `sticker_file_id` - A valid file identifier of the sticker.
32    pub fn new<A, B>(id: A, sticker_file_id: B) -> Self
33    where
34        A: Into<String>,
35        B: Into<String>,
36    {
37        Self {
38            id: id.into(),
39            sticker_file_id: sticker_file_id.into(),
40            input_message_content: None,
41            reply_markup: None,
42        }
43    }
44
45    /// Sets a new input message content.
46    ///
47    /// # Arguments
48    ///
49    /// * `value` - Content of the message to be sent instead of the photo.
50    pub fn with_input_message_content<T>(mut self, value: T) -> Self
51    where
52        T: Into<InputMessageContent>,
53    {
54        self.input_message_content = Some(value.into());
55        self
56    }
57
58    /// Sets a new reply markup.
59    ///
60    /// # Arguments
61    ///
62    /// * `value` - Reply markup.
63    pub fn with_reply_markup<T>(mut self, value: T) -> Self
64    where
65        T: Into<InlineKeyboardMarkup>,
66    {
67        self.reply_markup = Some(value.into());
68        self
69    }
70}

72impl TryFrom<RawInlineQueryResult> for InlineQueryResultCachedSticker {
73    type Error = RawInlineQueryResultDataError;
74
75    fn try_from(value: RawInlineQueryResult) -> Result<Self, Self::Error> {
76        Ok(Self {
77            id: value.id,
78            input_message_content: value.data.input_message_content,
79            reply_markup: value.data.reply_markup,
80            sticker_file_id: value.data.sticker_file_id.ok_or(MissingField("sticker_file_id"))?,
81        })
82    }
83}
