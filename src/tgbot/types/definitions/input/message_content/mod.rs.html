12/// Represents a content of a message to be sent as a result of an inline query.
13#[derive(Clone, Debug, derive_more::From, Deserialize, PartialEq, PartialOrd, Serialize)]
14#[serde(untagged)]
15pub enum InputMessageContent {
16    /// Represents a contact.
17    Contact(InputMessageContentContact),
18    /// Represents an invoice.
19    Invoice(InputMessageContentInvoice),
20    /// Represents a venue.
21    Venue(InputMessageContentVenue),
22    /// Represents a location.
23    Location(InputMessageContentLocation),
24    /// Represents a text.
25    Text(InputMessageContentText),
26}
27
28impl From<Contact> for InputMessageContent {
29    fn from(value: Contact) -> Self {

31    }
32}

34impl From<Location> for InputMessageContent {
