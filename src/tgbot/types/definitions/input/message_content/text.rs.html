1use serde::{Deserialize, Serialize};
2
3use crate::types::{LinkPreviewOptions, ParseMode, Text, TextEntities, TextEntity};
4
5/// Represents a text message to be sent as the result of an inline query.
6#[serde_with::skip_serializing_none]
7#[derive(Clone, Debug, Deserialize, PartialEq, PartialOrd, Serialize)]
8pub struct InputMessageContentText {
9    message_text: String,
10    entities: Option<TextEntities>,
11    link_preview_options: Option<LinkPreviewOptions>,
12    parse_mode: Option<ParseMode>,
13}
14
15impl InputMessageContentText {
16    /// Creates a new `InputMessageContentText`.
17    ///
18    /// # Arguments
19    ///
20    /// * `value` - Text; 1-4096 characters.
21    pub fn new<T>(value: T) -> Self
22    where
23        T: Into<String>,
24    {
25        Self {
26            message_text: value.into(),
27            entities: None,
28            link_preview_options: None,
29            parse_mode: None,
30        }
31    }
32
33    /// Sets a new list of entities.
34    ///
35    /// # Arguments
36    ///
37    /// * `value` - List of special entities that appear in the text.
38    ///
39    /// Parse mode will be set to [`None`] when this method is called.
40    pub fn with_entities<T>(mut self, value: T) -> Self
41    where
42        T: IntoIterator<Item = TextEntity>,
43    {
44        self.entities = Some(value.into_iter().collect());
45        self.parse_mode = None;
46        self
47    }
48
49    /// Sets a new link preview options.
50    ///
51    /// # Arguments
52    ///
53    /// * `value` - Link preview generation options for the message.
54    pub fn with_link_preview_options(mut self, value: LinkPreviewOptions) -> Self {
55        self.link_preview_options = Some(value);
56        self
57    }
58
59    /// Sets a new parse mode.
60    ///
61    /// # Arguments
62    ///
63    /// * `value` - Parse mode.
64    ///
65    /// Entities will be set to [`None`] when this method is called.
66    pub fn with_parse_mode(mut self, value: ParseMode) -> Self {
67        self.parse_mode = Some(value);
68        self.entities = None;
69        self
70    }
71}
72
73impl<T> From<T> for InputMessageContentText
74where
75    T: Into<String>,
76{
77    fn from(value: T) -> Self {
78        Self::new(value)
79    }
80}
81
82impl From<Text> for InputMessageContentText {
83    fn from(value: Text) -> Self {
84        let mut result = Self::new(value.data);
85        if let Some(entities) = value.entities {
86            result = result.with_entities(entities);
87        }
88        result
89    }
