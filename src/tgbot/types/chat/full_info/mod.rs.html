23/// Type of the chat.
24#[derive(Clone, Copy, Debug, Deserialize, PartialEq, PartialOrd, Serialize)]
25#[serde(rename_all = "snake_case")]
26pub enum ChatFullInfoType {
27    /// Channel chat.
28    Channel,
29    /// Group chat.
30    Group,
31    /// Private chat.
32    Private,
33    /// Supergroup chat.
34    Supergroup,
35}
36
37/// Contains full information about a chat.
38#[serde_with::skip_serializing_none]
39#[derive(Clone, Debug, Deserialize, PartialEq, Serialize)]
40pub struct ChatFullInfo {
41    /// Unique identifier for this chat.
42    pub id: Integer,
43    #[serde(rename = "type")]
44    /// Type of the chat.
45    pub chat_type: ChatFullInfoType,
46
47    /// Identifier of the accent color for the chat name and backgrounds of the chat photo, reply header, and link preview.
48    #[serde(rename = "accent_color_id")]
49    pub accent_color: AccentColor,
50    /// The maximum number of reactions that can be set on a message in the chat.
51    pub max_reaction_count: Integer,
52
53    /// List of available reactions allowed in the chat. If omitted, then all emoji reactions are allowed.
54    pub available_reactions: Option<Vec<ReactionType>>,
55    /// Custom emoji identifier of the emoji chosen by the chat for the reply header and link preview background.
56    pub background_custom_emoji_id: Option<String>,
57    /// Whether non-administrators can only get the list of bots and administrators in the chat.
58    pub has_hidden_members: Option<bool>,
59    /// Whether messages from the chat can't be forwarded to other chats.
60    pub has_protected_content: Option<bool>,
61    /// Whether new chat members will have access to old messages; available only to chat administrators.
62    pub has_visible_history: Option<bool>,
63    /// The time after which all messages sent to the chat will be automatically deleted; in seconds.
64    pub message_auto_delete_time: Option<Integer>,
65    /// Chat photo.
66    pub photo: Option<ChatPhoto>,
67    /// The most recent pinned message (by sending date).
68    pub pinned_message: Option<Message>,
69    /// Identifier of the accent color for the chat's profile background.
70    #[serde(rename = "profile_accent_color_id")]
71    pub profile_accent_color: Option<ProfileAccentColor>,
72    /// Custom emoji identifier of the emoji chosen by the chat for its profile background.
73    pub profile_background_custom_emoji_id: Option<String>,
74
75    /// The list of all active chat usernames; for private chats, supergroups and channels.
76    pub active_usernames: Option<Vec<String>>,
77    /// Username, for private chats, supergroups and channels.
78    pub username: Option<String>,
79
80    /// Description, for supergroups, channels and group chats.
81    pub description: Option<String>,
82    /// Primary invite link, for supergroups, channels and group chats.
83    pub invite_link: Option<String>,
84
85    /// Title, for supergroups, channels and group chats.
86    pub title: Option<String>,
87
88    /// Whether paid media messages can be sent or forwarded to the channel chat.
89    ///
90    /// The field is available only for channel chats.
91    pub can_send_paid_media: Option<bool>,
92
93    /// Unique identifier for the linked chat, i.e. the discussion group identifier for a channel and vice versa;
94    /// for supergroups and channel chats.
95    pub linked_chat_id: Option<Integer>,
96
97    /// Bio of the other party in a private chat.
98    pub bio: Option<String>,
99    /// For private chats, the date of birth of the user.
100    pub birthdate: Option<Birthdate>,
101    /// For private chats with business accounts, the intro of the business.
102    pub business_intro: Option<BusinessIntro>,
103    /// For private chats with business accounts, the location of the business.
104    pub business_location: Option<BusinessLocation>,
105    /// For private chats with business accounts, the opening hours of the business.
106    pub business_opening_hours: Option<BusinessOpeningHours>,
107    /// Custom emoji identifier of the emoji status of the chat or the other party in a private chat.
108    pub emoji_status_custom_emoji_id: Option<String>,
109    /// Expiration date of the emoji status of the chat or the other party in a private chat, in Unix time, if any.
110    pub emoji_status_expiration_date: Option<Integer>,
111    /// First name of the other party in a private chat.
112    pub first_name: Option<String>,
113    /// Whether privacy settings of the other party in the private chat
114    /// allows to use `tg://user?id=<user_id>` links only in chats with the user.
115    pub has_private_forwards: Option<bool>,
116    /// Whether the privacy settings of the other party restrict
117    /// sending voice and video note messages in the private chat.
118    pub has_restricted_voice_and_video_messages: Option<bool>,
119    /// Last name of the other party in a private chat.
120    pub last_name: Option<String>,
121    /// For private chats, the personal channel of the user.
122    pub personal_chat: Option<Chat>,
123
124    /// Whether the bot can change the group sticker set.
125    pub can_set_sticker_set: Option<bool>,
126    /// For supergroups, the name of the group's custom emoji sticker set.
127    /// Custom emoji from this set can be used by all users and bots in the group.
128    pub custom_emoji_sticker_set_name: Option<String>,
129    /// Whether aggressive anti-spam checks are enabled in the supergroup.
130    /// The field is only available to chat administrators.
131    pub has_aggressive_anti_spam_enabled: Option<bool>,
132    /// Whether the supergroup chat is a forum (has topics enabled).
133    pub is_forum: Option<bool>,
134    /// Whether users need to join the supergroup before they can send messages.
135    pub join_to_send_messages: Option<bool>,
136    /// Whether all users directly joining the supergroup need to be approved by supergroup administrators.
137    pub join_by_request: Option<bool>,
138    /// For supergroups, the location to which the supergroup is connected.
139    pub location: Option<ChatLocation>,
140    /// For supergroups, the minimum allowed delay between
141    /// consecutive messages sent by each unprivileged user; in seconds.
142    pub slow_mode_delay: Option<Integer>,
143    /// For supergroups, name of the group sticker set.
144    pub sticker_set_name: Option<String>,
145    /// For supergroups, the minimum number of boosts that a non-administrator
146    /// user needs to add in order to ignore slow mode and chat permissions.
147    pub unrestrict_boost_count: Option<Integer>,
148
149    /// Default chat member permissions, for groups and supergroups.
150    pub permissions: Option<ChatPermissions>,
151
152    /// Information about types of gifts that are accepted by the chat or by the corresponding user for private chats.
153    pub accepted_gift_types: Option<AcceptedGiftTypes>,
154}
