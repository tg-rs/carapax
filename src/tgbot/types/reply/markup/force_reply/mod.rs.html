6/// Represents a reply interface trigger.
7///
8/// Upon receiving a message with this object, Telegram clients
9/// will display a reply interface to the user.
10/// This can be extremely useful if you want to create user-friendly step-by-step interfaces
11/// without having to sacrifice [privacy mode][1].
12///
13/// [1]: https://core.telegram.org/bots/features#privacy-mode
14#[serde_with::skip_serializing_none]
15#[derive(Clone, Debug, Deserialize, PartialEq, PartialOrd, Serialize)]
16pub struct ForceReply {
17    force_reply: bool,
18    input_field_placeholder: Option<String>,
19    selective: Option<bool>,
20}
21
22impl ForceReply {
23    /// Creates a new `ForceReply`.
24    ///
25    /// # Arguments
26    ///
27    /// * `force_reply` - Indicates whether to show reply interface to the user,
28    ///   as if they manually selected the bot‘s message and tapped ’Reply'.
29    pub fn new(force_reply: bool) -> Self {
30        ForceReply {
31            force_reply,
32            input_field_placeholder: None,
33            selective: None,
34        }
35    }
36
37    /// Sets a new input field placeholder.
38    ///
39    /// # Arguments
40    ///
41    /// * `value` - The placeholder to be shown
42    ///   in the input field when
43    ///   the keyboard is active;
44    ///   1-64 characters.
45    pub fn with_input_field_placeholder<T>(mut self, value: T) -> Self
46    where
47        T: Into<String>,
48    {
49        self.input_field_placeholder = Some(value.into());
50        self
51    }
52
53    /// Sets a new value for the `selective` flag.
54    ///
55    /// # Arguments
56    ///
57    /// * `value` - Value of the flag.
58    ///
59    /// Use this parameter if you want to force reply from specific users only.
60    ///
61    /// Targets:
62    ///
63    /// 1. users that are `@mentioned` in the text of the Message object.
64    /// 2. if the bot message is a reply (has `reply_to_message_id`), sender of the original message.
65    pub fn with_selective(mut self, selective: bool) -> Self {
66        self.selective = Some(selective);
67        self
68    }
69}
70
71impl From<bool> for ForceReply {
72    fn from(value: bool) -> Self {
73        Self::new(value)
74    }
75}
