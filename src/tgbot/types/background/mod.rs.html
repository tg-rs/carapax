8/// Describes the way a background is filled based on the selected colors.
9#[derive(Clone, Debug, Deserialize, PartialEq, PartialOrd, Serialize)]
10#[serde(rename_all = "snake_case", tag = "type")]
11pub enum BackgroundFill {
12    /// The background is a freeform gradient that rotates after every message in the chat.
13    FreeformGradient {
14        /// A list of the 3 or 4 base colors that are used to generate the freeform gradient in the RGB24 format.
15        colors: Vec<Integer>,
16    },
17    /// The background is a gradient fill.
18    Gradient {
19        /// Bottom color of the gradient in the RGB24 format.
20        bottom_color: Integer,
21        /// Clockwise rotation angle of the background fill in degrees; 0-359.
22        rotation_angle: Integer,
23        /// Top color of the gradient in the RGB24 format.
24        top_color: Integer,
25    },
26    /// The background is filled using the selected color.
27    Solid {
28        /// The color of the background fill in the RGB24 format.
29        color: Integer,
30    },
31}

33/// This object describes the type of a background.
34#[serde_with::skip_serializing_none]
35#[derive(Clone, Debug, Deserialize, PartialEq, PartialOrd, Serialize)]
36#[serde(rename_all = "snake_case", tag = "type")]
37pub enum BackgroundType {
38    /// The background is taken directly from a built-in chat theme.
39    ChatTheme {
40        /// Name of the chat theme, which is usually an emoji.
41        theme_name: String,
42    },
43    /// The background is automatically filled based on the selected colors.
44    Fill {
45        /// Dimming of the background in dark themes, as a percentage; 0-100.
46        dark_theme_dimming: Integer,
47        /// The background fill.
48        fill: BackgroundFill,
49    },
50    /// The background is a PNG or TGV (gzipped subset of SVG with MIME type “application/x-tgwallpattern”)
51    /// pattern to be combined with the background fill chosen by the user.
52    Pattern {
53        /// Document with the pattern.
54        document: Document,
55        ///The background fill that is combined with the pattern.
56        fill: BackgroundFill,
57        /// Intensity of the pattern when it is shown above the filled background; 0-100.
58        intensity: Integer,
59        ///  Whether the background fill must be applied only to the pattern itself. All other pixels are black in this case. For dark themes only.
60        is_inverted: Option<bool>,
61        /// Whether the background moves slightly when the device is tilted.
62        is_moving: Option<bool>,
63    },
64    /// The background is a wallpaper in the JPEG format.
65    Wallpaper {
66        /// Dimming of the background in dark themes, as a percentage; 0-100.
67        dark_theme_dimming: Integer,
68        /// Document with the wallpaper.
69        document: Document,
70        /// Whether the wallpaper is downscaled to fit in a 450x450 square and then box-blurred with radius 12.
71        is_blurred: Option<bool>,
72        /// Whether the background moves slightly when the device is tilted.
73        is_moving: Option<bool>,
74    },
75}
