22/// A thread-safe (but not very performant) implementation of a keyed rate limiter state
23/// store using [`HashMap`].
24///
25/// The `HashMapStateStore` is the default state store in `std` when no other thread-safe
26/// features are enabled.
27pub type HashMapStateStore<K, S = DefaultHasher> = Mutex<HashMap<K, InMemoryState, S>>;
28
29impl<K: Hash + Eq + Clone, S: core::hash::BuildHasher> StateStore for HashMapStateStore<K, S> {
30    type Key = K;
31
32    fn measure_and_replace<T, F, E>(&self, key: &Self::Key, f: F) -> Result<T, E>
33    where
34        F: Fn(Option<Nanos>) -> Result<(T, Nanos), E>,
35    {
36        let mut map = self.lock();
37        if let Some(v) = (*map).get(key) {
38            // fast path: a rate limiter is already present for the key.
39            return v.measure_and_replace_one(f);
40        }
41        // not-so-fast path: make a new entry and measure it.
42        let entry = (*map).entry(key.clone()).or_default();
43        entry.measure_and_replace_one(f)
44    }
45}

47impl<K: Hash + Eq + Clone, S: core::hash::BuildHasher> ShrinkableKeyedStateStore<K>
48    for HashMapStateStore<K, S>
49{
50    fn retain_recent(&self, drop_below: Nanos) {
51        let mut map = self.lock();
52        map.retain(|_, v| !v.is_older_than(drop_below));
53    }
54
55    fn shrink_to_fit(&self) {
56        let mut map = self.lock();
57        map.shrink_to_fit();
58    }
59
60    fn len(&self) -> usize {
61        let map = self.lock();
62        (*map).len()
63    }
64    fn is_empty(&self) -> bool {
65        let map = self.lock();
66        (*map).is_empty()
67    }
68}

70/// # Keyed rate limiters - [`HashMap`]-backed with a default hasher
71impl<K, C> RateLimiter<K, HashMapStateStore<K>, C, NoOpMiddleware<C::Instant>>
72where
73    K: Hash + Eq + Clone,
74    C: clock::Clock,
75{
76    /// Constructs a new rate limiter with a custom clock, backed by a [`HashMap`] with the default hasher.
77    pub fn hashmap_with_clock(quota: Quota, clock: C) -> Self {
78        let state: HashMapStateStore<K> = HashMapStateStore::new(HashMap::default());
79        RateLimiter::new(quota, state, clock)
80    }
81}
