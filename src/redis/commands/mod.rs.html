2211/// Allows pubsub callbacks to stop receiving messages.
2212///
2213/// Arbitrary data may be returned from `Break`.
2214pub enum ControlFlow<U> {
2215    /// Continues.
2216    Continue,
2217    /// Breaks with a value.
2218    Break(U),
2219}
2220
2221/// The PubSub trait allows subscribing to one or more channels
2222/// and receiving a callback whenever a message arrives.
2223///
2224/// Each method handles subscribing to the list of keys, waiting for
2225/// messages, and unsubscribing from the same list of channels once
2226/// a ControlFlow::Break is encountered.
2227///
2228/// Once (p)subscribe returns Ok(U), the connection is again safe to use
2229/// for calling other methods.
2230///
2231/// # Examples
2232///
2233/// ```rust,no_run
2234/// # fn do_something() -> redis::RedisResult<()> {
2235/// use redis::{PubSubCommands, ControlFlow};
2236/// let client = redis::Client::open("redis://127.0.0.1/")?;
2237/// let mut con = client.get_connection()?;
2238/// let mut count = 0;
2239/// con.subscribe(&["foo"], |msg| {
2240///     // do something with message
2241///     assert_eq!(msg.get_channel(), Ok(String::from("foo")));
2242///
2243///     // increment messages seen counter
2244///     count += 1;
2245///     match count {
2246///         // stop after receiving 10 messages
2247///         10 => ControlFlow::Break(()),
2248///         _ => ControlFlow::Continue,
2249///     }
2250/// })?;
2251/// # Ok(()) }
2252/// ```
2253// TODO In the future, it would be nice to implement Try such that `?` will work
2254//      within the closure.
2255pub trait PubSubCommands: Sized {
2256    /// Subscribe to a list of channels using SUBSCRIBE and run the provided
2257    /// closure for each message received.
2258    ///
2259    /// For every `Msg` passed to the provided closure, either
2260    /// `ControlFlow::Break` or `ControlFlow::Continue` must be returned. This
2261    /// method will not return until `ControlFlow::Break` is observed.
2262    fn subscribe<C, F, U>(&mut self, _: C, _: F) -> RedisResult<U>
2263    where
2264        F: FnMut(Msg) -> ControlFlow<U>,
2265        C: ToRedisArgs;

2267    /// Subscribe to a list of channels using PSUBSCRIBE and run the provided

2286    where
2287        F: FnMut(Msg) -> ControlFlow<U>,
2288        C: ToRedisArgs,
2289    {
2290        let mut pubsub = self.as_pubsub();

2296                ControlFlow::Continue => continue,
2297                ControlFlow::Break(value) => return Ok(value),
