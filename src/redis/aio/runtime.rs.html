51impl TaskHandle {

53    pub(crate) fn detach(self) {
54        match self {
55            #[cfg(feature = "smol-comp")]
56            TaskHandle::Smol(task) => task.detach(),
57            #[cfg(any(feature = "tokio-comp", feature = "async-std-comp"))]
58            _ => {}
59        }
60    }

71impl Drop for HandleContainer {
72    fn drop(&mut self) {
73        match self.0.take() {
74            None => {}
75            #[cfg(feature = "tokio-comp")]
76            Some(TaskHandle::Tokio(handle)) => handle.abort(),
77            #[cfg(feature = "async-std-comp")]
78            Some(TaskHandle::AsyncStd(handle)) => {
79                // schedule for cancellation without waiting for result.
80                // TODO - can we cancel the task without awaiting its completion?
81                async_std::task::spawn(async move { handle.cancel().await.unwrap_or_default() });
82            }
83            #[cfg(feature = "smol-comp")]
84            Some(TaskHandle::Smol(task)) => drop(task),
85        }
86    }
87}

130fn set_runtime(runtime: Runtime) -> Result<(), RedisError> {
131    const PREFER_RUNTIME_ERROR: &str =
132    "Another runtime preference was already set. Please call this function before any other runtime preference is set.";
133
134    CHOSEN_RUNTIME
135        .set(runtime)
136        .map_err(|_| RedisError::from((crate::ErrorKind::ClientError, PREFER_RUNTIME_ERROR)))
137}
138
139/// Mark Smol as the preferred runtime.
140///
141/// If the function returns `Err`, another runtime preference was already set, and won't be changed.
142/// Call this function if the application doesn't use multiple runtimes,
143/// but the crate is compiled with multiple runtimes enabled, which is a bad pattern that should be avoided.
144#[cfg(all(
145    feature = "smol-comp",
146    any(feature = "async-std-comp", feature = "tokio-comp")
147))]
148pub fn prefer_smol() -> Result<(), RedisError> {
149    set_runtime(Runtime::Smol)
150}
151
152/// Mark async-std compliant runtimes, such as smol, as the preferred runtime.
153///
154/// If the function returns `Err`, another runtime preference was already set, and won't be changed.
155/// Call this function if the application doesn't use multiple runtimes,
156/// but the crate is compiled with multiple runtimes enabled, which is a bad pattern that should be avoided.
157#[cfg(all(
158    feature = "async-std-comp",
159    any(feature = "tokio-comp", feature = "smol-comp")
160))]
161pub fn prefer_async_std() -> Result<(), RedisError> {
162    set_runtime(Runtime::AsyncStd)
163}
164
165/// Mark Tokio as the preferred runtime.
166///
167/// If the function returns `Err`, another runtime preference was already set, and won't be changed.
168/// Call this function if the application doesn't use multiple runtimes,
169/// but the crate is compiled with multiple runtimes enabled, which is a bad pattern that should be avoided.
170#[cfg(all(
171    feature = "tokio-comp",
172    any(feature = "async-std-comp", feature = "smol-comp")
173))]
174pub fn prefer_tokio() -> Result<(), RedisError> {
175    set_runtime(Runtime::Tokio)

178impl Runtime {
179    pub(crate) fn locate() -> Self {
180        #[cfg(any(
181            all(
182                feature = "tokio-comp",
183                any(feature = "async-std-comp", feature = "smol-comp")
184            ),
185            all(
186                feature = "smol-comp",
187                any(feature = "async-std-comp", feature = "tokio-comp")
188            ),
189            all(
190                feature = "async-std-comp",
191                any(feature = "tokio-comp", feature = "smol-comp")
192            )
193        ))]
194        if let Some(runtime) = CHOSEN_RUNTIME.get() {
195            return *runtime;
196        }
197
198        #[cfg(all(
199            feature = "tokio-comp",
200            not(feature = "async-std-comp"),
201            not(feature = "smol-comp")
202        ))]
203        {
204            Runtime::Tokio
205        }
206
207        #[cfg(all(
208            not(feature = "tokio-comp"),
209            not(feature = "smol-comp"),
210            feature = "async-std-comp"
211        ))]
212        {
213            Runtime::AsyncStd
214        }
215
216        #[cfg(all(
217            not(feature = "tokio-comp"),
218            feature = "smol-comp",
219            not(feature = "async-std-comp")
220        ))]
221        {
222            Runtime::Smol
223        }
224
225        cfg_if::cfg_if! {
226        if #[cfg(all(feature = "tokio-comp", feature = "async-std-comp"))] {
227            if ::tokio::runtime::Handle::try_current().is_ok() {
228                Runtime::Tokio
229            } else {
230                Runtime::AsyncStd
231            }
232        } else if #[cfg(all(feature = "tokio-comp", feature = "smol-comp"))] {
233            if ::tokio::runtime::Handle::try_current().is_ok() {
234                Runtime::Tokio
235            } else {
236                Runtime::Smol
237            }
238        } else if #[cfg(all(feature = "smol-comp", feature = "async-std-comp"))]
239            {
240                Runtime::AsyncStd
241            }
242        }
243
244        #[cfg(all(
245            not(feature = "tokio-comp"),
246            not(feature = "async-std-comp"),
247            not(feature = "smol-comp")
248        ))]
249        {
250            compile_error!(
251                "tokio-comp, async-std-comp, or smol-comp features required for aio feature"
252            )
253        }
254    }

256    #[must_use]
257    pub(crate) fn spawn(&self, f: impl Future<Output = ()> + Send + 'static) -> TaskHandle {
258        match self {
259            #[cfg(feature = "tokio-comp")]
260            Runtime::Tokio => crate_tokio::Tokio::spawn(f),
261            #[cfg(feature = "async-std-comp")]
262            Runtime::AsyncStd => crate_async_std::AsyncStd::spawn(f),
263            #[cfg(feature = "smol-comp")]
264            Runtime::Smol => crate_smol::Smol::spawn(f),
265        }
266    }

268    pub(crate) async fn timeout<F: Future>(
269        &self,
270        duration: Duration,
271        future: F,
272    ) -> Result<F::Output, Elapsed> {
273        match self {
274            #[cfg(feature = "tokio-comp")]
275            Runtime::Tokio => tokio::time::timeout(duration, future)
276                .await
277                .map_err(|_| Elapsed(())),
278            #[cfg(feature = "async-std-comp")]
279            Runtime::AsyncStd => async_std::future::timeout(duration, future)
280                .await
281                .map_err(|_| Elapsed(())),
282            #[cfg(feature = "smol-comp")]
283            Runtime::Smol => future.timeout(duration).await.ok_or(Elapsed(())),
284        }
285    }
286
287    #[cfg(any(feature = "connection-manager", feature = "cluster-async"))]
288    pub(crate) async fn sleep(&self, duration: Duration) {
289        match self {
290            #[cfg(feature = "tokio-comp")]
291            Runtime::Tokio => {
292                tokio::time::sleep(duration).await;
293            }
294            #[cfg(feature = "async-std-comp")]
295            Runtime::AsyncStd => {
296                async_std::task::sleep(duration).await;
297            }
298            #[cfg(feature = "smol-comp")]
299            Runtime::Smol => {
300                smol::Timer::after(duration).await;
301            }
302        }
303    }
304
305    #[cfg(feature = "cluster-async")]
306    pub(crate) async fn locate_and_sleep(duration: Duration) {
307        Self::locate().sleep(duration).await
308    }
